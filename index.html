// 1. Inisialisasi Peta dengan koordinat tengah wilayah Indonesia const map = L.map('map').setView([-2.5, 118.0], 5); // 2. Menambahkan Basemap Grid Standar (Basemap open-source sebagai latar) L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', { attribution: '© OpenStreetMap contributors' }).addTo(map); // 3. STREAMING MAP: Menghubungkan ke ArcGIS REST Service (Contoh data sampel Esri) // Jika menggunakan Ina-Geoportal, URL ini diganti dengan endpoint internal tanahair.indonesia.go.id const featureServiceUrl = 'https://arcgis.com'; // Tampilkan layer langsung ke peta secara dinamis (streaming vektor) const geoserverLayer = L.esri.featureLayer({ url: featureServiceUrl, style: function () { return { color: '#0079c1', weight: 2, fillOpacity: 0.2 }; } }).addTo(map); // Memberikan popup informasi ketika poligon peta di-klik pengguna geoserverLayer.bindPopup(function (layer) { return L.Util.template('

{STATE_NAME}

Total Populasi: {POP_CENSUS2020}

', layer.feature.properties); }); geoserverLayer.on('load', function() { document.getElementById('layer-status').innerText = "Layer berhasil dimuat dari server."; }); // 4. UNDUH DATA: Mengambil data spasial asli dari server menggunakan kueri REST API document.getElementById('download-btn').addEventListener('click', function () { document.getElementById('layer-status').innerText = "Sedang memproses unduhan..."; // Membuat URL kueri untuk meminta format mentah GeoJSON // `where=1=1` mengambil semua data, `f=geojson` memaksa server mengembalikan format berkas standar geospasial const queryUrl = `${featureServiceUrl}/query?where=1=1&outFields=*&f=geojson`; fetch(queryUrl) .then(response => { if (!response.ok) throw new Error("Gagal mengambil data dari server ArcGIS."); return response.json(); }) .then(geojsonData => { // Konversi objek JSON menjadi file text blob agar bisa diunduh browser const dataStr = "data:text/json;charset=utf-8," + encodeURIComponent(JSON.stringify(geojsonData)); const downloadAnchor = document.createElement('a'); downloadAnchor.setAttribute("href", dataStr); downloadAnchor.setAttribute("download", "data_rbi_extracted.geojson"); document.body.appendChild(downloadAnchor); downloadAnchor.click(); // Memicu download otomatis pada browser komputer downloadAnchor.remove(); document.getElementById('layer-status').innerText = "Unduhan selesai!"; }) .catch(error => { console.error(error); document.getElementById('layer-status').innerText = "Gagal mengunduh data."; alert("Terjadi kesalahan sistem saat meminta ekstraksi file."); }); });
